445 public:
446  FlagBit(unsigned short b)
447  : bitmask_(0)
448  {
449  if (b > 0) {
450  bitmask_ = 1;
451  bitmask_ = (bitmask_ << (b-1));
452  }
453  }
454 
455  bool isSet(const uint64_t& m) const {
456  return ((m & bitmask_)>0);
457  }
458 
459  uint64_t bitmask_;
460
